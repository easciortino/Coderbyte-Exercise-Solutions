function SwapCase(str) { 
    
    var stringArray = str.split("");
    
    return str.split("").map(function(element){
        if(element === element.toUpperCase()){
            return element.toLowerCase();
        }else{
            return element.toUpperCase();
        }        
    }).join("");;
    
    
  return str; 
         
}
